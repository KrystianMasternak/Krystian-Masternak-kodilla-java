SELECT A.SUMARY, B.NAME
FROM ISSUES A
JOIN ISSUESLISTS B ON A.ISSUESLISTS_ID = B.ID;

SELECT C.SUMARY, D.FIRSTNAME, D.LASTNAME
FROM ISSUES C
JOIN USERS D ON C.USER_ID_ASSIGNEDTO = D.ID;

SELECT D.FIRSTNAME, D.LASTNAME, COUNT(*) AS SUMARY
FROM USERS D JOIN ISSUES C ON D.ID = C.USER_ID_ASSIGNEDTO
GROUP BY C.USER_ID_ASSIGNEDTO
ORDER BY D.FIRSTNAME, D.LASTNAME;
